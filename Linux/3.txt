3) На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403

cd html #Перехожу в директорию html, где находится файл 404.html
cp 404.html 403.html #Копирую файл 404.html и переименовую на 403.html
sed -i 's/404/403/g' 403.html #Меняю в содержимом упоминания 404 на 403
